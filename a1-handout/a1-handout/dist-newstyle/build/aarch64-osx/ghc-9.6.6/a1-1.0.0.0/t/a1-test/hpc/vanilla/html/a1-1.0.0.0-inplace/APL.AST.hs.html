   10 data Exp
   11   = CstInt Integer
   12   | CstBool Bool
   13   | Add Exp Exp
   14   | Sub Exp Exp
   15   | Mul Exp Exp
   16   | Div Exp Exp
   17   | Pow Exp Exp
   18   | Eql Exp Exp
   19   | If Exp Exp Exp
   20   | Var VName

   27 printExp :: Exp -> String
   28 printExp (CstInt i) = show i 
   29 printExp (CstBool i) = show i
   30 printExp (Var name) = name
   31 printExp (Add e1 e2) = "(" ++ printExp e1 ++ "+" ++ printExp e2 ++ ")"
   32 printExp (Sub e1 e2) = "(" ++ printExp e1 ++ "-" ++ printExp e2 ++ ")"
   33 printExp (Mul e1 e2) = "(" ++ printExp e1 ++ "*" ++ printExp e2 ++ ")"
   34 printExp (Div e1 e2) = "(" ++ printExp e1 ++ "/" ++ printExp e2 ++ ")"
   35 printExp (Pow e1 e2) = "(" ++ printExp e1 ++ "**" ++ printExp e2 ++ ")"
   36 printExp (Eql e1 e2) = "(" ++ printExp e1 ++ "==" ++ printExp e2 ++ ")"
   37 printExp (If e1 e2 e3) = "(if " ++ printExp e1 ++ " then " ++ printExp e2 ++ " else " ++ printExp e3 ++ ")"
   38 printExp (Let var e1 e2) = "(" ++ "Let " ++ var ++ "=" ++ printExp e1 ++ " in " ++ printExp e2 ++ ")"
   39 printExp (Lambda name e1) = "(\\" ++ name ++ "->" ++ printExp e1 ++ ")"
   40 printExp (Apply e1 e2) = printExp e1 ++ " " ++ printExp e2
   41 printExp (TryCatch e1 e2) = "(try " ++ printExp e1 ++ " catch " ++ printExp e2 ++ ")"
