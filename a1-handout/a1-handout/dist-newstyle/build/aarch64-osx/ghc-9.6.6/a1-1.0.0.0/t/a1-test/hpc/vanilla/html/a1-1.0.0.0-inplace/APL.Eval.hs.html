   11 data Val
   12   = ValInt Integer
   13   | ValBool Bool
   14   | ValFun Env VName Exp
   15   deriving (Eq, Show)
   16 
   17 type Env = [(VName, Val)]
   18 
   19 envEmpty :: Env
   20 envEmpty = []
   21 
   22 envExtend :: VName -> Val -> Env -> Env
   23 envExtend v val env = (v, val) : env
   24 
   25 envLookup :: VName -> Env -> Maybe Val
   26 envLookup v env = lookup v env
   27 
   28 type Error = String
   29 
   30 evalIntBinOp :: (Integer -> Integer -> Either Error Integer) -> Env -> Exp -> Exp -> Either Error Val
   31 evalIntBinOp f env e1 e2 =
   32   case (eval env e1, eval env e2) of
   33     (Left err, _) -> Left err
   34     (_, Left err) -> Left err
   35     (Right (ValInt x), Right (ValInt y)) -> case f x y of
   36       Left err -> Left err
   37       Right z -> Right $ ValInt z
   38     (Right _, Right _) -> Left "Non-integer operand"
   39 
   40 evalIntBinOp' :: (Integer -> Integer -> Integer) -> Env -> Exp -> Exp -> Either Error Val
   41 evalIntBinOp' f env e1 e2 =
   42   evalIntBinOp f' env e1 e2
   43   where
   44     f' x y = Right $ f x y
   45 
   46 eval :: Env -> Exp -> Either Error Val
   47 eval _env (CstInt x) = Right $ ValInt x
   48 eval _env (CstBool b) = Right $ ValBool b
   49 eval env (Var v) = case envLookup v env of
   50   Just x -> Right x
   51   Nothing -> Left $ "Unknown variable: " ++ v
   52 eval env (Add e1 e2) = evalIntBinOp' (+) env e1 e2
   53 eval env (Sub e1 e2) = evalIntBinOp' (-) env e1 e2
   54 eval env (Mul e1 e2) = evalIntBinOp' (*) env e1 e2
   55 eval env (Div e1 e2) = evalIntBinOp checkedDiv env e1 e2
   56   where
   57     checkedDiv _ 0 = Left "Division by zero"
   58     checkedDiv x y = Right $ x `div` y
   59 eval env (Pow e1 e2) = evalIntBinOp checkedPow env e1 e2
   60   where
   61     checkedPow x y =
   62       if y < 0
   63         then Left "Negative exponent"
   64         else Right $ x ^ y
   65 eval env (Eql e1 e2) =
   66   case (eval env e1, eval env e2) of
   67     (Left err, _) -> Left err
   68     (_, Left err) -> Left err
   69     (Right (ValInt x), Right (ValInt y)) -> Right $ ValBool $ x == y
   70     (Right (ValBool x), Right (ValBool y)) -> Right $ ValBool $ x == y
   71     (Right _, Right _) -> Left "Invalid operands to equality"
   72 eval env (If cond e1 e2) =
   73   case eval env cond of
   74     Left err -> Left err
   75     Right (ValBool True) -> eval env e1
   76     Right (ValBool False) -> eval env e2
   77     Right _ -> Left "Non-boolean conditional."
   78 eval env (Let var e1 e2) =
   79   case eval env e1 of
   80     Left err -> Left err
   81     Right v -> eval (envExtend var v env) e2
   82 -- Task: Functions
   83 eval env (Lambda var e1) = 
   84   Right (ValFun env var e1)
   85 eval env (Apply funE argE) = 
   86   case eval env funE of
   87     Left err -> Left err
   88     Right (ValFun env2 name e2) -> 
   89       case eval env argE of 
   90         Left err -> Left err
   91         Right argV -> eval (envExtend name argV env2) e2
   92     Right _ -> Left "error non ValFun"
   93 -- Task: try-catch
   94 eval env (TryCatch e1 e2) = 
   95   case eval env e1 of
   96     Right x -> Right x
   97     Left _ -> eval env e2
